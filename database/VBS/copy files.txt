'============================'Copy files between folders'============================Dim sOriginFolder, sDestinationFolder, sFile, oFSOSet oFSO = CreateObject("Scripting.FileSystemObject")sOriginFolder = "system.app"sDestinationFolder = "CD\Output\data\"For Each sFile In oFSO.GetFolder(sOriginFolder).FilesIf Not oFSO.FileExists(sDestinationFolder & "\" & oFSO.GetFileName(sFile)) ThenoFSO.GetFile(sFile).Copy sDestinationFolder & "\" & oFSO.GetFileName(sFile),TrueEnd ifnextWScript.Echo "Copying :  All the files have been copied to " & sDestinationFolder '================================================================'if file does not exist in folder then copy file from source OR'if file does exist, overwrite existing file'================================================================Dim sFilesToCopy(1)sFilesToCopy(0) = "system.app\Walker.exe"sFilesToCopy(1) = "system.app\Walker.swf"sDestinationFolder = "CD\Output\data\"Const OverwriteExisting = TRUESet objFSO = CreateObject("Scripting.FileSystemObject")For Each sFile In sFilesToCopyIf objFSO.FolderExists(sDestinationFolder) Then  objFSO.CopyFile sFile, sDestinationFolder, OverwriteExisting  Else  Wscript.Echo "Target Folder does not exist."End IfnextWScript.Echo "Copying Files:" & vbNewline & Join(sFilesToCopy) & vbNewline & "have been copied to " & sDestinationFolder'Dim msg'msg = "VBScript SecondLine"'msg=Split (msg," ")'Mynewstr= msg(0) & vbNewline 'This will create an newline'Mynewstr=Mynewstr & msg(1)'MsgBox Mynewstr
